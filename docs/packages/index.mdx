# 패키지

이 색인은 우리의 모노레포에서 관리되는 패키지들을 나열합니다.
여기에는 우리의 핵심 컴파일러인 `@mdx-js/mdx`, MDX 구문을 지원하는 remark 플러그인인 `remark-mdx`, 그리고 번들러와 프론트엔드 프레임워크를 위한 여러 통합 패키지들이 포함됩니다.

{
  (function () {
    const navigationTree = props.navigationTree
    const category = navigationTree.children.find(function (item) {
      return item.name === '/packages/'
    })
    assert(category)

    return (
      <nav>
        <NavigationGroup items={category.children} includeDescription />
      </nav>
    )
  })()
}
